--show databases, drop database (if it exists), and create new database
SHOW DATABASES;
DROP DATABASE IF EXISTS oabdelqaassign2db;
CREATE DATABASE IF NOT EXISTS oabdelqaassign2db;

--show databases again to confirm successful creation
SHOW DATABASES; 

--connect to database
USE oabdelqaassign2db

--grant access to TA for marking 
GRANT USAGE ON *.* TO 'ta'@'localhost';
DROP USER 'ta'@'localhost';
CREATE USER 'ta'@'localhost' IDENTIFIED BY 'cs3319';
GRANT ALL PRIVILEGES ON oabdelqaassign2db.* TO 'ta'@'localhost';
FLUSH PRIVILEGES;

--list all tables 
SHOW TABLES; 

--create 4 tables needed for database

CREATE TABLE product (productid VARCHAR(2) NOT NULL, description 
VARCHAR(20), price INT, quantityOnHand INT, PRIMARY KEY(productid));

CREATE TABLE agent (agentid VARCHAR(2) NOT NULL, firstname VARCHAR(30), 
lastname VARCHAR(30), city VARCHAR(15), commission INT, PRIMARY 
KEY(agentid));

CREATE TABLE customer (customerid VARCHAR(2) NOT NULL, firstname 
VARCHAR(30), lastname VARCHAR(30), city VARCHAR(15), phoneNum 
INT, agentid VARCHAR(2), PRIMARY KEY(customerid), FOREIGN KEY(agentid) 
REFERENCES agent(agentid));

CREATE TABLE purchases (customerid VARCHAR(2), productid VARCHAR(2), 
quantitypurchased INT, PRIMARY KEY(customerid, productid), FOREIGN 
KEY(customerid) REFERENCES customer(customerid), FOREIGN KEY(productid) 
REFERENCES product(productid)); 

--list all tables to confirm creation 
SHOW TABLES;
